<configuration>
  <springProperty scope="context" name="APP_NAME" source="spring.application.name"/>
  <springProperty scope="context" name="ENV"      source="spring.application.env"/>

  <!-- real (synchronous) appender -->
  <appender name="CONSOLE" class="ch.qos.logback.core.ConsoleAppender">
    <layout class="ch.qos.logback.classic.PatternLayout">
      <pattern>%d{yyyy-MM-dd HH:mm:ss.SSS} [%thread] %-5level %logger{36}
        app.id=${APP_NAME} env=${ENV} correlation.id=%X{x-correlation-id:-''} %msg%n
      </pattern>
    </layout>
  </appender>

  <!-- async wrapper -->
  <appender name="ASYNC_CONSOLE" class="ch.qos.logback.classic.AsyncAppender">
    <!-- Tune these to your traffic -->
    <queueSize>8192</queueSize>                <!-- ring buffer size -->
    <discardingThreshold>0</discardingThreshold><!-- 0 = don't drop DEBUG/TRACE under load -->
    <neverBlock>true</neverBlock>              <!-- return immediately; drop when full -->
    <!-- if you prefer back-pressure instead of dropping, set neverBlock to false -->
    <appender-ref ref="CONSOLE"/>
  </appender>

  <!-- your package/category -->
  <logger name="com.dbs.cb" level="INFO" additivity="false">
    <appender-ref ref="ASYNC_CONSOLE"/>
  </logger>

  <!-- root goes to async console as well -->
  <root level="INFO">
    <appender-ref ref="ASYNC_CONSOLE"/>
  </root>
</configuration>
